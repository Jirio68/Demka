Как сделать авторизацию по ролям
Часть 1:
	Изначально у БД должна быть таблица по типу "users" в котором есть столбцы "ID_пользователя", "Логин", "Пароль", "Роль"
	
	1)Создаём класс через "Проект" -> "Создать класс..." и называем "PersonalData" (для удобства)
	2)и в этом классе вставляем этот код:
	
	public static int IdUser { get; private set; }
	public static string Password { get; private set; }
	public static string Login { get; private set; }
	public static string Role { get; private set; }

public bool SetPersonalData(string login, string password)
{
    var db = new DB();
    string sqlExpression = "Select TOP 1 * FROM Пользователи where Логин  = @login and Пароль = @Password";
    {
        using (SqlConnection connection = new SqlConnection(db.stringCon()))
        {
            connection.Open();
            using (SqlCommand command = new SqlCommand(sqlExpression, connection))
            {
                command.Parameters.AddWithValue("@Login", login);
                command.Parameters.AddWithValue("@Password", password);

                using (SqlDataReader reader = command.ExecuteReader())
                {
                    if (reader.HasRows)
                    {
                        reader.Read();
                        IdUser = (int)reader["ID_Пользователя"];
                        Login = reader["Логин"].ToString();
                        Password = reader["Пароль"].ToString();
                        Role = reader["Роль"].ToString();

                        return true;

                    }
                }
                return false;
            }
        }
    }
}

Как вывести МИНИМАЛЬНОЕ МАКСИМАЛЬНОЕ СРЕДНЕЕ значение для какого-то столбца в DataGridView
1) Добавить DataGridView (не добавляя "источник данных") и кнопку
2) Двойной щелчок по кнопке и добавить этот код:

	DB db = new DB();
	string connectionString = db.stringCon();

	// SQL-запрос
	string query = @"
		SELECT 
			MIN(PricePerCopy) AS MinPrice,
			MAX(PricePerCopy) AS MaxPrice,
			ROUND(AVG(PricePerCopy), 2) AS AvgPrice
		FROM Books";

	try
	{
		using (SqlConnection connection = new SqlConnection(connectionString))
		{
			SqlDataAdapter adapter = new SqlDataAdapter(query, connection);
			DataTable dataTable = new DataTable();
			adapter.Fill(dataTable);

			// Привязываем DataTable к DataGridView
			dataGridView1.DataSource = dataTable;

			// Изменяем заголовки столбцов
			dataGridView1.Columns["MinPrice"].HeaderText = "Минимальная цена";
			dataGridView1.Columns["MaxPrice"].HeaderText = "Максимальная цена";
			dataGridView1.Columns["AvgPrice"].HeaderText = "Средняя цена";
		}
	}
	catch (Exception ex)
	{
		MessageBox.Show("Ошибка: " + ex.Message);
	}

Важно! вместо "PricePerCopy" пишем нужный столбец, а вместо "Books" нужную таблицу
